# argodeploy/applications.yaml v1.3
apiVersion: argoproj.io/v1alpha1
kind: ApplicationSet
metadata:
  name: gitops-poc-applications
  namespace: argocd
spec:
  generators:
    - git:
        repoURL: 'https://github.com/andres20980/gitops-poc.git'
        revision: HEAD
        directories:
          - path: apps/*
  template:
    metadata:
      name: '{{path.basename}}-app' # Ej: pdf-processor-app, world-app
      labels:
        app.kubernetes.io/managed-by: argocd
        app.kubernetes.io/part-of: gitops-poc
    spec:
      project: default
      source:
        repoURL: 'oci://ghcr.io/andres20980/chart-base' # Apunta directamente al chart-base remoto
        targetRevision: 1.2.1 # La versión específica del chart-base
        # NOTA: Se elimina la línea 'chart: chart-base'
        helm:
          valueFiles:
            - '{{path}}/common-values.yaml' # Añadido en v1.2 si no lo tenias
            - '{{path}}/values.yaml'
            - '{{path}}/values-dev.yaml'
            - '{{path}}/values-pre.yaml' # Añadido en v1.2 si no lo tenias
      destination:
        server: 'https://kubernetes.default.svc'
        namespace: '{{path.basename}}'
      syncPolicy:
        automated:
          prune: true
          selfHeal: true
        syncOptions:
          - CreateNamespace=true
